  9 require_relative 'path'
 10 
 11 class Maze
 12   attr_reader :grid
 13 
 14   def initialize(grid = nil)
 15     @grid = grid || [[Square.new(' ')]]
 16   end
 17 
 18   def [](x, y)
 19     grid[y][x]
 20   end
 21 
 22   def start
 23     begin
 24       find_pos(:start?)
 25     rescue
 26       throw "Failed call to Maze#start: this maze does not have a start"
 27     end
 28   end
 29 
 30   def read_text(file)
 31     @grid = text_to_grid(read_file(file))
 32   end
 33 
 34   def x_bound
 35     grid.length
 36   end
 37 
 38   def y_bound
 39     grid[0].length
 40   end
 41 
 42   def to_s
 43     grid.map { |row| print_row(row) }.join("\n")
 44   end
 45 
 46   def print
 47     grid.map { |row| print_with_marks(row) }.join("\n")
 48   end
 49 
 50   def ==(other)
 51     grid == other.grid
 52   end
 53 
 54   def out_of_bounds?(pos)
 55     x, y = pos
 56     x < 0 || x_bound <= x || y < 0 || y_bound <= y
 57   end
 58 
 59   def reset_marks
 60     each { |square| square.unmark }
 61   end
 62 
 63   private
 64 
 65   def print_row(row)
 66     row.map { |square| square.to_s }.join
 67   end
 68 
 69   def print_with_marks(row)
 70     row.map { |square| square.print }.join
 71   end
 72 
 73   def text_to_grid(text)
 74     text.split("\n").map do |row|
 75       row.chars.map { |char| Square.new(char) }
 76     end
 77   end
 78 
 79   def read_file(file)
 80     File.read(file)
 81   end
 82 
 83   def find_pos(type)
 84     grid.each_with_index do |row, y|
 85       row.each_with_index { |square, x| return [x, y] if square.send(type) }
 86     end
 87 
 88     raise "A position of type #{type} could not be found"
 89   end
 90 
 91   def each
 92     grid.flatten.each { |square| yield(square) }
 93   end
 94 
 95   class Square
 96     attr_reader :content
 97 
 98     def initialize(content)
 99       @content = content
100       @marked = false
101     end
102 
103     def mark
104       @marked = true
105       self
106     end
107 
108     def unmark
109       @mark = false
110     end
111 
112     def wall?
113       content == '*'
114     end
115 
116     def start?
117       content == 'S'
118     end
119 
120     def finish?
121       content == 'E'
122     end
123 
124     def empty?
125       !self.wall?
126     end
127 
128     def marked?
129       @marked
130     end
131 
132     def ==(other)
133       content == other.content
134     end
135 
136     def to_s
137       content
138     end
139 
140     def print
141       marked? ? 'X' : self.to_s
142     end
143   end
144 end
